JavaScript // ========================================== // FILE: Code.gs // MỤC ĐÍCH: Xử lý API Request (Đăng nhập, Lưu dữ liệu) // ========================================== /** * Hàm mặc định của GAS để nhận POST request từ web Frontend */ function doPost(e) { // Khai báo header CORS để cho phép gọi API từ GitHub Pages var headers = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "POST, GET, OPTIONS", "Access-Control-Allow-Headers": "Content-Type" }; try { // Parse dữ liệu JSON từ Frontend gửi lên var payload = JSON.parse(e.postData.contents); var action = payload.action; var responseData = {}; // Bộ định tuyến (Router) if (action === "LOGIN") { responseData = handleLogin(payload.data); } else if (action === "SAVE_DATA") { responseData = handleSaveData(payload.data); } else { responseData = { status: "error", message: "Action không hợp lệ!" }; } // Trả về kết quả dạng JSON return ContentService.createTextOutput(JSON.stringify(responseData)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.message })).setMimeType(ContentService.MimeType.JSON); } } /** * Xử lý logic Đăng nhập (Đối chiếu với Sheet "Users") */ function handleLogin(data) { var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var sheet = ss.getSheetByName(CONFIG.SHEET_USERS); var records = sheet.getDataRange().getValues(); var mssvInput = data.mssv.toString().trim(); var passInput = data.password.toString().trim(); // Duyệt từ dòng 2 (bỏ qua Header) for (var i = 1; i < records.length; i++) { var mssvDb = records[i][0].toString().trim(); var passDb = records[i][1].toString().trim(); if (mssvDb === mssvInput && passDb === passInput) { return { status: "success", message: "Đăng nhập thành công", user: { mssv_id: mssvDb, full_name: records[i][2], role: records[i][3], group_id: records[i][4] } }; } } return { status: "error", message: "Sai MSSV hoặc mật khẩu!" }; } /** * Xử lý lưu dữ liệu bài thực hành & Upload Ảnh */ function handleSaveData(data) { var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var processedSheet = ss.getSheetByName(CONFIG.SHEET_PROCESSED); var rawSheet = ss.getSheetByName(CONFIG.SHEET_RAW); var timestamp = new Date(); // access_datetime // 1. XỬ LÝ UPLOAD ẢNH LÊN GOOGLE DRIVE (Nếu có Base64) var individualPhotoUrl = ""; var groupPhotoUrl = ""; if (data.individual_photo_base64) { individualPhotoUrl = saveImageToDrive(data.individual_photo_base64, data.student_id + "_ind_Buoi" + data.session_name); } if (data.group_photo_base64) { groupPhotoUrl = saveImageToDrive(data.group_photo_base64, data.group_id + "_grp_Buoi" + data.session_name); } // 2. LƯU DỮ LIỆU VÀO SHEET "Processed_Data" // Map đúng 14 cột theo thiết kế Database ban đầu var processedRow = [ timestamp, // A: access_datetime data.session_name, // B: session_name data.submit_type, // C: submit_type data.student_id, // D: student_id data.student_name, // E: student_name data.group_id, // F: group_id data.machine_type || "", // G: machine_type data.target_name || "", // H: target_name JSON.stringify(data.result_avg), // I: result_avg (Gom các biến TB) data.qc_evaluation || "", // J: qc_evaluation data.f_h_calc || "", // K: f_h_calc individualPhotoUrl, // L: individual_photo_url groupPhotoUrl, // M: group_photo_url data.student_comment || "" // N: student_comment ]; processedSheet.appendRow(processedRow); // 3. LƯU DỮ LIỆU VÀO SHEET "Raw_Data_Log" (Audit 3 lần đo) // Map 6 cột theo thiết kế var rawRow = [ timestamp, // A: access_datetime data.session_name, // B: session_name data.student_id, // C: student_id JSON.stringify(data.r1_data), // D: r1_data JSON.stringify(data.r2_data), // E: r2_data JSON.stringify(data.r3_data) // F: r3_data ]; rawSheet.appendRow(rawRow); return { status: "success", message: "Dữ liệu và hình ảnh đã được lưu trữ thành công." }; } /** * Hàm Hỗ trợ: Lưu chuỗi Base64 thành file ảnh trên Google Drive */ function saveImageToDrive(base64Data, fileName) { try { // Tách phần data khỏi chuỗi prefix (VD: "data:image/jpeg;base64,...") var dataString = base64Data.split(',')[1] || base64Data; var blob = Utilities.newBlob(Utilities.base64Decode(dataString), MimeType.JPEG, fileName + ".jpg"); var folder = DriveApp.getFolderById(CONFIG.IMAGE_FOLDER_ID); var file = folder.createFile(blob); // Set quyền chia sẻ để web có thể render thẻ file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); // Trả về link trực tiếp để hiển thị ảnh return "https://drive.google.com/uc?export=view&id=" + file.getId(); } catch (e) { return "Lỗi upload ảnh: " + e.message; } } Các bước triển khai (Deployment) trên Google Apps Script: Mở file Google Sheets DB_ThucHanhTracDia.
